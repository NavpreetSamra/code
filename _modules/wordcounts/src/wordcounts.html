[docs]    def words_tokenizer(txt):
        """
        Tokenize with nltk removing punctuations

        :param str txt: text to toknize
        :return: tokens
        :rtype: list
        """
        return [word.lower() for word in
                nltk.wordpunct_tokenize(txt) if word.isalpha()]
