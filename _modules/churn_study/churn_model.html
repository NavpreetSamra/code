[docs]class Cleaner(object):
    """
    Clean and reorganize data

    * convert times to datetime dtype
    * remove users < 30 days (None in current data)
    * move phone to bool
    * (diff in driver rider ratings potentially)
    * isnull flag for avg_ratings, then replace with mean
    * create churn flag
    * split city

    :param str fpath: path to file to generate DataFrame
    :param pandas.DataFrame data: DataFrame of data
    :param array-like timeCols: str time columns to convert to date time
    """
    def __init__(self, fpath=None, data=None, timeCols=['last_trip_date', 'signup_date']):
        """
        """
        if not data:
            self.df = pd.read_csv(fpath)
        else:
            self.df = data
        self.timeCols = timeCols

        self.clean_ratings()
        self.normalize_floats()
        self.clean_car()
        self.clean_phone()
        self.convert_time()
        self.combine_surge()
        self.create_churn()
        self.split_city()


[docs]    def clean_ratings(self):
        """
        Create not null flags for avg ratings, then replace with mean val

        * by driver mean=4.77 std=.45
        * of driver mean=4.60 std=.61

        Point of future work move this to post city split
        """
        notBy = pd.isnull(self.df.avg_rating_by_driver)
        notOf = pd.isnull(self.df.avg_rating_of_driver)
        byMean = self.df.avg_rating_by_driver.mean()
        ofMean = self.df.avg_rating_of_driver.mean()
        self.df['notBy'] = notBy.astype('int')
        self.df['notOf'] = notOf.astype('int')
        self.df.avg_rating_by_driver[notBy] = byMean
        self.df.avg_rating_of_driver[notOf] = ofMean

[docs]    def clean_phone(self):
        """
        iPhone=1, Android=0, nan=.5
        """
        pval = .5 * np.ones(self.df.shape[0])
        pval[self.df.phone.values == "iPhone"] = 1.
        pval[self.df.phone.values == "Android"] = 0.
        self.df['phone'] = pval

[docs]    def split_city(self):
        """
        Split by city

        * King's Landing 62.8
        * Winterfell 35.2
        * Astapor 25.5
        """
        self.city = {}
        self.city['all'] = {}
        self.city['all']['df'] = self.df
        for i in set(self.df.city):
            self.city[i] = {}
            mask = self.df.city == i
            self.city[i]['df'] = self.df.ix[mask]

[docs]def col_churn(col, churn):
    """
    calculate churn rate based on boolean column, with conversion if nec.

    :param str col: column name of column in DataFrame
    """
    if col.dtype == float:
        mask = np.logical_or(col == 1., col == 0.)
        col = col[mask].astype('int').astype('bool')
        churn = churn[mask]
    col = col.astype('bool')
    pos = float(churn[col].sum()) / len(churn[col])
    neg = float(churn[np.invert(col)].sum()) / len(churn[np.invert(col)])
    return (neg, pos, sum(col)/float(len(col)) )
